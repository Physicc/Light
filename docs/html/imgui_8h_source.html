689  // The typical call flow is:
690  // - 1. Call BeginTable().
691  // - 2. Optionally call TableSetupColumn() to submit column name/flags/defaults.

694  // - 5. Populate contents:
695  // - In most situations you can use TableNextRow() + TableSetColumnIndex(N) to start appending into a column.
696  // - If you are using tables as a sort of grid, where every columns is holding the same type of contents,
697  // you may prefer using TableNextColumn() instead of TableNextRow() + TableSetColumnIndex().
698  // TableNextColumn() will automatically wrap-around into the next row if needed.
699  // - IMPORTANT: Comparatively to the old Columns() API, we need to call TableNextColumn() for the first column!

701  // --------------------------------------------------------------------------------------------------------
702  // TableNextRow() -> TableSetColumnIndex(0) -> Text("Hello 0") -> TableSetColumnIndex(1) -> Text("Hello 1") // OK
703  // TableNextRow() -> TableNextColumn() -> Text("Hello 0") -> TableNextColumn() -> Text("Hello 1") // OK
704  // TableNextColumn() -> Text("Hello 0") -> TableNextColumn() -> Text("Hello 1") // OK: TableNextColumn() automatically gets to next row!

1700 //-----------------------------------------------------------------------------
1701 // ImVector<>
1702 // Lightweight std::vector<>-like class to avoid dragging dependencies (also, some implementations of STL with debug enabled are absurdly slow, we bypass it so our code runs fast in debug).
1703 //-----------------------------------------------------------------------------
1704 // - You generally do NOT need to care or use this ever. But we need to make it available in imgui.h because some of our public structures are relying on it.
1705 // - We use std-like naming convention here, which is a little unusual for this codebase.
1706 // - Important: clear() frees memory, resize(0) keep the allocated buffer. We use resize(0) a lot to intentionally recycle allocated buffers across frames and amortize our costs.
1707 // - Important: our implementation does NOT call C++ constructors/destructors, we treat everything as raw data! This is intentional but be extra mindful of that,
1708 // Do NOT use this class as a std::vector replacement in your own code! Many of the structures used by dear imgui can be safely initialized by a zero-memset.
1709 //-----------------------------------------------------------------------------
1710 
1711 template<typename T>
1712 struct ImVector
1713 {
1714  int Size;
1715  int Capacity;
1716  T* Data;
1717 
1718  // Provide standard typedefs but we don't use them ourselves.
1719  typedef T value_type;
1720  typedef value_type* iterator;
1721  typedef const value_type* const_iterator;
1722 
1723  // Constructors, destructor
1724  inline ImVector() { Size = Capacity = 0; Data = NULL; }
1725  inline ImVector(const ImVector<T>& src) { Size = Capacity = 0; Data = NULL; operator=(src); }

1727  inline ~ImVector() { if (Data) IM_FREE(Data); }
1728 
1729  inline bool empty() const { return Size == 0; }
1730  inline int size() const { return Size; }
1731  inline int size_in_bytes() const { return Size * (int)sizeof(T); }
1732  inline int max_size() const { return 0x7FFFFFFF / (int)sizeof(T); }
1733  inline int capacity() const { return Capacity; }
1734  inline T& operator[](int i) { IM_ASSERT(i >= 0 && i < Size); return Data[i]; }
1735  inline const T& operator[](int i) const { IM_ASSERT(i >= 0 && i < Size); return Data[i]; }
1736 
1737  inline void clear() { if (Data) { Size = Capacity = 0; IM_FREE(Data); Data = NULL; } }
1738  inline T* begin() { return Data; }
1739  inline const T* begin() const { return Data; }
1740  inline T* end() { return Data + Size; }
1741  inline const T* end() const { return Data + Size; }
1742  inline T& front() { IM_ASSERT(Size > 0); return Data[0]; }
1743  inline const T& front() const { IM_ASSERT(Size > 0); return Data[0]; }
1744  inline T& back() { IM_ASSERT(Size > 0); return Data[Size - 1]; }
1745  inline const T& back() const { IM_ASSERT(Size > 0); return Data[Size - 1]; }
1746  inline void swap(ImVector<T>& rhs) { int rhs_size = rhs.Size; rhs.Size = Size; Size = rhs_size; int rhs_cap = rhs.Capacity; rhs.Capacity = Capacity; Capacity = rhs_cap; T* rhs_data = rhs.Data; rhs.Data = Data; Data = rhs_data; }

1748  inline int _grow_capacity(int sz) const { int new_capacity = Capacity ? (Capacity + Capacity / 2) : 8; return new_capacity > sz ? new_capacity : sz; }
1749  inline void resize(int new_size) { if (new_size > Capacity) reserve(_grow_capacity(new_size)); Size = new_size; }
1750  inline void resize(int new_size, const T& v) { if (new_size > Capacity) reserve(_grow_capacity(new_size)); if (new_size > Size) for (int n = Size; n < new_size; n++) memcpy(&Data[n], &v, sizeof(v)); Size = new_size; }
1751  inline void shrink(int new_size) { IM_ASSERT(new_size <= Size); Size = new_size; } // Resize a vector to a smaller size, guaranteed not to cause a reallocation
1752  inline void reserve(int new_capacity) { if (new_capacity <= Capacity) return; T* new_data = (T*)IM_ALLOC((size_t)new_capacity * sizeof(T)); if (Data) { memcpy(new_data, Data, (size_t)Size * sizeof(T)); IM_FREE(Data); } Data = new_data; Capacity = new_capacity; }
1753 
1754  // NB: It is illegal to call push_back/push_front/insert with a reference pointing inside the ImVector data itself! e.g. v.push_back(v[10]) is forbidden.
1755  inline void push_back(const T& v) { if (Size == Capacity) reserve(_grow_capacity(Size + 1)); memcpy(&Data[Size], &v, sizeof(v)); Size++; }
1756  inline void pop_back() { IM_ASSERT(Size > 0); Size--; }
1757  inline void push_front(const T& v) { if (Size == 0) push_back(v); else insert(Data, v); }
1758  inline T* erase(const T* it) { IM_ASSERT(it >= Data && it < Data + Size); const ptrdiff_t off = it - Data; memmove(Data + off, Data + off + 1, ((size_t)Size - (size_t)off - 1) * sizeof(T)); Size--; return Data + off; }
1759  inline T* erase(const T* it, const T* it_last){ IM_ASSERT(it >= Data && it < Data + Size && it_last > it && it_last <= Data + Size); const ptrdiff_t count = it_last - it; const ptrdiff_t off = it - Data; memmove(Data + off, Data + off + count, ((size_t)Size - (size_t)off - count) * sizeof(T)); Size -= (int)count; return Data + off; }
1760  inline T* erase_unsorted(const T* it) { IM_ASSERT(it >= Data && it < Data + Size); const ptrdiff_t off = it - Data; if (it < Data + Size - 1) memcpy(Data + off, Data + Size - 1, sizeof(T)); Size--; return Data + off; }
1761  inline T* insert(const T* it, const T& v) { IM_ASSERT(it >= Data && it <= Data + Size); const ptrdiff_t off = it - Data; if (Size == Capacity) reserve(_grow_capacity(Size + 1)); if (off < (int)Size) memmove(Data + off + 1, Data + off, ((size_t)Size - (size_t)off) * sizeof(T)); memcpy(&Data[off], &v, sizeof(v)); Size++; return Data + off; }
1762  inline bool contains(const T& v) const { const T* data = Data; const T* data_end = Data + Size; while (data < data_end) if (*data++ == v) return true; return false; }
1763  inline T* find(const T& v) { T* data = Data; const T* data_end = Data + Size; while (data < data_end) if (*data == v) break; else ++data; return data; }
1764  inline const T* find(const T& v) const { const T* data = Data; const T* data_end = Data + Size; while (data < data_end) if (*data == v) break; else ++data; return data; }
1765  inline bool find_erase(const T& v) { const T* it = find(v); if (it < Data + Size) { erase(it); return true; } return false; }
1766  inline bool find_erase_unsorted(const T& v) { const T* it = find(v); if (it < Data + Size) { erase_unsorted(it); return true; } return false; }
1767  inline int index_from_ptr(const T* it) const { IM_ASSERT(it >= Data && it < Data + Size); const ptrdiff_t off = it - Data; return (int)off; }
1768 };
